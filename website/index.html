wallet.rs 
use pqcrypto_dilithium::dilithium5::*;
use sha3::{Sha3_256, Digest};

pub struct Wallet {
    pub address: String,
    pub public_key: PublicKey,
    secret_key: SecretKey,
}

impl Wallet {
    pub fn new() -> Self {
        let (pk, sk) = keypair();
        let mut hasher = Sha3_256::new();
        hasher.update(pk.as_bytes());
        let hash = hasher.finalize();
        let address = format!(
            "CUR{}", hex::encode(&hash[..20])
        );

        Wallet { address, public_key: pk, secret_key: sk }
    }

    pub fn sign(&self, msg: &[u8]) -> SignedMessage {
        detached_sign(msg, &self.secret_key)
    }
}
REST API

Build on CURS3D with a simple HTTP API

10 endpoints with full CORS support. Query the chain, look up accounts, submit transactions -- all from your browser or any HTTP client.

GET /api/status Chain height, finalized height, validators, pending txs
GET /api/blocks Recent blocks with pagination (?from=0&limit=20)
GET /api/block/:height Full block detail with transactions, state root, merkle root
GET /api/account/:addr Balance, nonce, and staked balance for any address
GET /api/tx/:hash Transaction details by hash (kind, from, to, amount, fee)
GET /api/pending All pending transactions in the mempool
GET /api/validators Active validator set with addresses, public keys, and stakes
GET /api/faucet/:addr Testnet faucet -- instantly credits 100 CUR
POST /api/tx/submit Submit a signed transaction (JSON body)
OPTIONS * CORS preflight -- all origins, methods, headers allowed
